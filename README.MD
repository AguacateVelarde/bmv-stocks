## Bolsa mexicana de valores

```js
const { loadAllStocks } = require("bmv-stocks");
loadAllStocks().then((stocks) => {
  fs.writeFileSync("./stocks.json", JSON.stringify(stocks), function (
    err,
    data
  ) {
    if (err) {
      return console.log(err);
    }
  });

  process.exit(0);
});
```
